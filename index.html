// ========== 卡片5：7天做市奖励（修复版） ========== // 查询用户累计领取总额（通过事件） async function getTotalClaimedRewards(userAddress) { if (!web3 || !userAddress) return 0; try { const contract = new web3.eth.Contract(REWARD_POOL_ABI, RP); const events = await contract.getPastEvents('DailyRewardClaimed', { filter: { user: userAddress }, fromBlock: 0, toBlock: 'latest' }); let total = 0; events.forEach(event => { total += parseFloat(Web3.utils.fromWei(event.returnValues.amount, 'ether')); }); return total; } catch(e) { console.warn('获取累计领取失败', e); return 0; } } // 查询今日预计可领金额 async function getTodayExpectedReward(userAddress) { if (!web3 || !userAddress) return 0; try { const mykBalance = await getMYKBalance(userAddress); const rateRaw = await web3.eth.call({ to: RP, data: encodeCall('getRoleRate(address)', ['address'], [userAddress]) }); const rate = parseInt(web3.eth.abi.decodeParameter('uint256', rateRaw)) / 10; // 10→1.0% return (mykBalance * rate) / 100; } catch(e) { return 0; } } // 检查今日是否已领取 async function hasClaimedToday(userAddress) { if (!web3 || !userAddress) return false; try { const todayStart = Math.floor(new Date().setHours(0,0,0,0) / 1000); const contract = new web3.eth.Contract(REWARD_POOL_ABI, RP); const events = await contract.getPastEvents('DailyRewardClaimed', { filter: { user: userAddress }, fromBlock: 0, toBlock: 'latest' }); const todayEvent = events.find(e => parseInt(e.returnValues.timestamp) >= todayStart); return !!todayEvent; } catch(e) { return false; } } // 更新卡片5显示 async function updateCard5() { if (!userAddress) return; // 并行查询所有数据 const [buyValue, streak, totalClaimed, todayExpected, claimedToday, isBuyer] = await Promise.all([ getLastPancakeBuyValue(userAddress), // 最近购买USDT价值 getStreakLength(userAddress), // 连续天数 getTotalClaimedRewards(userAddress), // 累计已领 getTodayExpectedReward(userAddress), // 今日预计可领 hasClaimedToday(userAddress), // 今日是否已领 isRecentPancakeBuyer(userAddress) // 7天内有买入 ]); // 更新DOM $('holdingBalance').textContent = buyValue.toFixed(2) + ' USDT'; $('streakDays').textContent = streak; $('totalClaimed').textContent = totalClaimed.toFixed(4) + ' MYK'; $('todayRewardAmount').textContent = todayExpected.toFixed(4) + ' MYK'; // 按钮状态 const canClaim = isBuyer && !claimedToday; $('claimRewardBtn').disabled = !canClaim; $('claimRewardBtn').textContent = canClaim ? '📈 领取今日奖励' : (!isBuyer ? '🔒 需7天内有买入' : '✅ 今日已领取'); }
