const { MerkleTree } = require('merkletreejs'); const keccak256 = require('keccak256'); // 每天定时运行 async function generateDailyMerkleTree() { // 1. 获取所有活跃用户的持币量 const holders = await getActiveHolders(); // [{address, balance}] // 2. 计算每人今日奖励 const rewards = holders.map(h => ({ address: h.address, amount: calculateReward(h.balance) })); // 3. 生成叶子节点 const leaves = rewards.map(r => keccak256( ethers.utils.solidityPack( ['address', 'uint256'], [r.address, ethers.utils.parseEther(r.amount)] ) ) ); // 4. 构建 Merkle Tree const tree = new MerkleTree(leaves, keccak256, { sortPairs: true }); const root = tree.getHexRoot(); // 5. 上传 Root 到合约（需要管理员权限） // await contract.setMerkleRoot(root, batchId); // 6. 返回每个用户的 Proof const proofs = {}; for (const reward of rewards) { const leaf = keccak256( ethers.utils.solidityPack( ['address', 'uint256'], [reward.address, ethers.utils.parseEther(reward.amount)] ) ); proofs[reward.address] = { amount: reward.amount, proof: tree.getHexProof(leaf), root: root, batchId: getTodayBatchId() }; } return proofs; }
